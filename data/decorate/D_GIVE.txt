actor "Menu" : Weapon
{
  obituary "%o was killed."
  inventory.pickupmessage "Snagged an item!"  
  weapon.selectionorder 500
  weapon.kickback 0
  decal BulletChip
  +NOAUTOFIRE
  +NOALERT
  states
  {
  Spawn:
    TNT1 A 0
    stop
  Ready: 
    TNT1 A 0
	TNT1 A 0 ACS_ExecuteAlways(635,0,0,0,0)
    TNT1 A 0 ACS_ExecuteAlways(636,0,1,0,0)
    TNT1 A 0 A_WeaponReady
    TNT1 A 1 A_ChangeFlag("PICKUP",0)
    goto Ready + 3
  Deselect:
    TNT1 A 0 A_Lower
    TNT1 A 0 A_SelectWeapon("Menu")
    TNT1 A 0 A_Lower
    TNT1 A 1 A_Lower
    loop
  Select:
    TNT1 A 0 A_Raise
    TNT1 A 0 A_Raise
    TNT1 A 1 A_Raise
    loop
  Fire:
    TNT1 A 1 A_GiveInventory("MenuCursor",1)
    TNT1 A 0 ACS_ExecuteAlways(636,0,0,0,0)
	goto Ready+3
  AltFire:
    TNT1 A 1 ACS_ExecuteAlways(637,0,0,0,0)
    goto Ready+3
  }
}

actor "MenuChoice1" : Weapon
{
  obituary "%o was killed."
  inventory.pickupmessage "Snagged an item!"  
  weapon.selectionorder 500
  weapon.kickback 0
  decal BulletChip
  +NOAUTOFIRE
  +NOALERT
  states
  {
  Spawn:
    TNT1 A 0
    stop
  Ready:
    TNT1 A 0
	TNT1 A 0 A_TakeInventory("MenuCursor", 20)
	//goto GiveCursor
  GiveCursor:
	TNT1 A 0
	goto FinishReady
  FinishReady:
    TNT1 A 0 ACS_ExecuteAlways(637,0,0,0,0)
    TNT1 A 1 A_ChangeFlag("PICKUP",0)
    TNT1 A 1 A_SelectWeapon("Menu")
	TNT1 A 0 A_WeaponReady
	goto FinishReady+1
  Deselect:
	TNT1 A 0 A_Lower
	TNT1 A 0 A_Lower
	TNT1 A 0 A_Lower
	TNT1 A 0 A_Lower
	TNT1 A 0 A_Lower
    TNT1 A 0 A_Lower
    loop
  Select:
	TNT1 A 0 A_Raise
	TNT1 A 0 A_Raise
	TNT1 A 0 A_Raise
	TNT1 A 0 A_Raise
	TNT1 A 0 A_Raise
    loop
  Fire:
    TNT1 A 1 
	goto Ready
  }
}
